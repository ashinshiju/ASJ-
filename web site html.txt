  <script>
    const correctPassword = "universal@01"; // Updated password

    function promptPassword(section) {
      const password = prompt("Enter the password to access this section:");
      
      if (password === correctPassword) {
        if (section === 'personal') {
          toggleUpload();
        } else if (section === 'hacks') {
          displayMaintenance();
        }
      } else {
        alert("Incorrect password!");
      }
    }

    function toggleUpload() {
      const uploadSection = document.getElementById('upload-section');
      uploadSection.style.display = uploadSection.style.display === 'none' ? 'block' : 'none';
    }

    function displayMaintenance() {
      const maintenanceMessage = document.getElementById('maintenance-message');
      maintenanceMessage.style.display = 'block';
      setTimeout(() => {
        alert("Hacker's Touch is currently under maintenance.");
      }, 1000);
    }

    const fileInput = document.getElementById('file-upload');
    const storageInfo = document.getElementById('storage-info');
    const MAX_STORAGE_MB = 20480; // 20 GB

    fileInput.addEventListener('change', function () {
      let totalSize = 0;
      for (const file of fileInput.files) {
        totalSize += file.size;
      }
      const totalMB = totalSize / (1024 * 1024);
      storageInfo.textContent = `Storage used: ${totalMB.toFixed(2)} MB / ${MAX_STORAGE_MB} MB`;

      if (totalMB > MAX_STORAGE_MB) {
        alert('You have exceeded the 20 GB limit. Please remove some files.');
        fileInput.value = '';
        storageInfo.textContent = `Storage used: 0 MB / ${MAX_STORAGE_MB} MB`;
      }
    });
  </script>
